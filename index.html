  <script>

    // ✅ SETEAZĂ AICI DATA DE LAUNCH (timezone local)

    // Exemplu: 2026-02-01 12:00

    const LAUNCH_DATE = new Date("2026-02-01T12:00:00");


    // year

    document.getElementById("year").textContent = new Date().getFullYear();


    // copy link

    function copyLink(e){

      e.preventDefault();

      navigator.clipboard?.writeText(location.href).then(()=>{

        const el = e.currentTarget;

        const old = el.textContent;

        el.textContent = "Copied ✅";

        setTimeout(()=> el.textContent = old, 1200);

      });

    }


    // countdown

    const $d = document.getElementById("d");

    const $h = document.getElementById("h");

    const $m = document.getElementById("m");

    const $s = document.getElementById("s");


    function tick(){

      const now = new Date();

      let diff = LAUNCH_DATE - now;


      if(diff <= 0){

        $d.textContent = "0";

        $h.textContent = "0";

        $m.textContent = "0";

        $s.textContent = "0";

        return;

      }


      const sec = Math.floor(diff / 1000);

      const days = Math.floor(sec / 86400);

      const hours = Math.floor((sec % 86400) / 3600);

      const mins = Math.floor((sec % 3600) / 60);

      const secs = sec % 60;


      $d.textContent = String(days);

      $h.textContent = String(hours).padStart(2,"0");

      $m.textContent = String(mins).padStart(2,"0");

      $s.textContent = String(secs).padStart(2,"0");

    }

    tick();

    setInterval(tick, 1000);


    // starfield (no external assets)

    const canvas = document.getElementById("stars");

    const ctx = canvas.getContext("2d");


    let W, H, DPR;

    let stars = [];


    function resize(){

      DPR = Math.min(window.devicePixelRatio || 1, 2);

      W = canvas.width = Math.floor(window.innerWidth * DPR);

      H = canvas.height = Math.floor(window.innerHeight * DPR);

      canvas.style.width = window.innerWidth + "px";

      canvas.style.height = window.innerHeight + "px";


      const count = Math.floor((window.innerWidth * window.innerHeight) / 14000);

      stars = Array.from({length: count}, () => ({

        x: Math.random() * W,

        y: Math.random() * H,

        r: (Math.random() * 1.2 + 0.4) * DPR,

        v: (Math.random() * 0.35 + 0.08) * DPR,

        a: Math.random() * 0.6 + 0.2

      }));

    }


    function draw(){

      ctx.clearRect(0,0,W,H);

      ctx.globalCompositeOperation = "lighter";


      for(const st of stars){

        st.y += st.v;

        if(st.y > H + 10){ st.y = -10; st.x = Math.random() * W; }


        ctx.beginPath();

        ctx.fillStyle = `rgba(255,255,255,${st.a})`;

        ctx.arc(st.x, st.y, st.r, 0, Math.PI * 2);

        ctx.fill();

      }


      ctx.globalCompositeOperation = "source-over";

      requestAnimationFrame(draw);

    }


    window.addEventListener("resize", resize);

    resize();

    draw();

  </script>
